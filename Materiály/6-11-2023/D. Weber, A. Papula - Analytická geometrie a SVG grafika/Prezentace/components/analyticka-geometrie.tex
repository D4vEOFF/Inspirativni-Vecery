\section{Analytická geometrie}

\subsection{Lineární zobrazení (homomorfismus)}
\begin{frame}[t]{Lineární zobrazení}
    \only<2-3>{
        \begin{block}{Definice homomorfismu}
            Jsou-li $U,V$ vektorové prostory nad tělesem $T$, pak zobrazení $\mapping{f}{U}{V}$ je \emph{lineární (homomorfismus)}, pokud platí
            \begin{enumerate}[I]
                \item \(\forall x,y\in U:\,f(x+y)=f(x)+f(y),\)
                \item \(\forall x\in V,\forall \alpha\in T:\,f(\alpha x)=\alpha\cdot f(x).\)
            \end{enumerate}
        \end{block}
    }
    \only<3>{
        Nás bude zajímat především prostor $\R^2$, nebot' budeme pracovat s rovinou $\implies U=V=\R^2$.
    }
    \only<4>{
        \begin{block}{Vyjádření homomorfismu}
            Každý homomorfismus $\mapping{f}{U}{V}$ lze vyjádřit ve tvaru $f(x)=Ax$, kde
            \[A=\left(\begin{matrix}
                | & & |\\
                f(v_1) & \cdots & f(v_n)\\
                | & & |
            \end{matrix}\right),\]
            přičemž $\set{v_1,\dots,v_n}$ je báze prostoru $U$, tj. $\mspan{U}$.
        \end{block}
    }
\end{frame}

\subsection{Afinní zobrazení}
\begin{frame}[t]{Afinní zobrazení}
    \only<1->{
        \begin{block}{Defininice afinního zobrazení}
            Buď $\mapping{g}{U}{V}$ homomorfismus a $b\in V$ vektor. Pak \emph{afinní zobrazení} $\mapping{f}{U}{V}$ má tvar $f(x)=g(x)+b$, kde $x\in U$.
        \end{block}
    }
    \only<2->{
        Afinní zobrazení $f$ lze tedy zapsat ve tvaru
    \[f(x)=Ax+b,\]
    kde $A$ je matice homomorfismu $g$.
    }
\end{frame}

\subsection{Homogenní souřadnice}
\begin{frame}[t]{Homogenní souřadnice}
    \only<2->{
        \begin{itemize}
            \only<2->{\item Lze využít tzv. \textbf{homogenních souřadnic} $\implies$ rozšíříme prostor o jednu dimenzi.}
            \only<3->{\item Obecný vektor $u=\T{(u_1,u_2)}\in \R^2$ zapíšeme jako $\T{(u_1,u_2,1)}$}
            \only<4->{\item Díky tomu můžeme posunutí o vektor $b$ v daném afinním zobrazení $f$ zapsat jako \emph{pouhé} násobení matice s vektorem:}
            \only<5->{
                \[\left(\begin{matrix}
                    x_1^\prime\\
                    x_2^\prime\\
                    1
                \end{matrix}\right)=\only<5>{\left(\begin{matrix}
                    a_{11} & a_{21} & b_1\\
                    a_{12} & a_{22} & b_2\\
                    0 & 0 & 1
                \end{matrix}\right)}\only<6->{\underbrace{\left(\begin{matrix}
                    a_{11} & a_{21} & b_1\\
                    a_{12} & a_{22} & b_2\\
                    0 & 0 & 1
                \end{matrix}\right)}_{\text{Afinní matice}}}\cdot\left(\begin{matrix}
                    x_1\\
                    x_2\\
                    1
                \end{matrix}\right).\]
            }
            \only<7->{\item To odpovídá původnímu zápisu
            \[f(x)=\left(\begin{matrix}
                x_1^\prime\\
                x_2^\prime
            \end{matrix}\right)=Ax+b\]}
        \end{itemize}
    }
\end{frame}

\subsection{Skládání afinních zobrazení}
\begin{frame}[t]{Skládání afinních zobrazení}
    \only<2->{
        \begin{block}{Matice složeného zobrazení}
            Jsou-li $\mapping{f}{U}{V}$ a $\mapping{g}{V}{W}$ afinní zobrazení, kde matice přislušných homomorfismů jsou po řadě $A,B$, pak matice homomorfismu složeného afinního zobrazení $f\circ g$ je $AB$.
        \end{block}
    }
    \only<3->{
        O tom se lze přesvědčit:
    \[(f\circ g)(x)=f(g(x))=A\overbrace{(Bx+v)}^{g(x)}+u=(AB)x+(Av+u).\]
    }
    \only<4->{
        Pro \textbf{afinní} matice platí totéž $\implies$ maticí složeného afinního zobrazení je \textbf{součin afinních matic}.
    }
\end{frame}

\begin{frame}[t]{Příklad složeného zobrazení}
    \begin{itemize}
        \item Mějme afinní zobrazení $f,g$ po řadě s afinními maticemi $\mathcal{T}(\T{(3,2)})$, $\mathcal{R}(\pi/2)$ $\implies$ \textbf{translace} a \textbf{rotace}.
        \item Matice zobrazení $f\circ g$:
        \begin{align*}
            \mathcal{T}(\T{(3,2)})\cdot\mathcal{R}\left(\dfrac{\pi}{2}\right)&=\left(\begin{matrix}
                1 & 0 & 3\\
                0 & 1 & 2\\
                0 & 0 & 1
            \end{matrix}\right)\cdot\left(\begin{matrix}
                \cos(\pi/2) & -\sin(\pi/2) & 0\\
                \sin(\pi/2) & \cos(\pi/2) & 0\\
                0 & 0 & 1
            \end{matrix}\right)\\
            &=\left(\begin{matrix}
                0 & -1 & 3\\
                1 & 0 & 2\\
                0 & 0 & 1
            \end{matrix}\right)
        \end{align*}
    \end{itemize}
\end{frame}
